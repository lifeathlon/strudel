setcpm(105/4)

samples({
  krim: 'krim.wav',
}, 'https://raw.githubusercontent.com/lifeathlon/strudel/main/samples/stuff/')


var darkDrone = x => x
  .sound("supersaw")
  .transpose(-24)
  .slow(8)
  .speed(.35)
  .attack(4)
  .release(8)
  .room(1)
  .delay(.15)
  .delaytime(.75)
  .delayfeedback(.6)
  .lpf(450)
  .lpq(10)
  .gain(.55)


var dreamy = x => x
  .speed(.75)
  .delay(.35)
  .delaytime(.375)
  .delayfeedback(.45)
  .room(.8)
  .gain(.8)
  .slow(1)

var meditative = x => x
  .speed(1)
  .attack(.4)
  .release(2.5)
  .delay(.25)
  .delaytime(.5)
  .delayfeedback(.35)
  .room(1)
  .lpf(1400)
  .gain(.65)
  .slow(1)

var krim = meditative(s("<krim*1 _>"))
$: darkDrone(n("<0>").scale("a:minor")).gain(.2)
._pianoroll()
$: stack(
  krim,
  
  )
._pianoroll()
